
PARAMETERS: 
 STRING HAKTEXT,
 STRING SOURCELANG,
 STRING TARGETLANG;

OBJECT: 
 STRING HAKRESPONSE,
 STRING DATAREQUEST,
 TABLE TRANSX,
 STRING INPUT,
 STRING OUTPUT;

SELECT '' AS LANG,'' AS TEXT 
	FROM IASBAS001 
	WHERE 1 = 2 
	INTO HAKTBLTRANS;

APPEND ROW TO HAKTBLTRANS;

IF STRLEN(HAKTEXT) == 0 
		|| STRLEN(HAKTEXT) == 0 
		|| STRLEN(HAKTEXT) == 0 THEN
	HAKRESPONSE = 'PARAMETRELER BOŞ OLAMAZ';
	RETURN HAKRESPONSE;
ELSE
	DATAREQUEST =   '{"mtext":'+TOCHAR(34)+ HAKTEXT+TOCHAR(34) + ',"text":[""],"source_lang":'+TOCHAR(34)+ SOURCELANG+TOCHAR(34) + ',"target_lang":'+TOCHAR(34)+ TARGETLANG+TOCHAR(34) + '}';
	SENDHTTPPOST DATAREQUEST TO 'http://xx.xx.xx.xx:yyyy/translate' CODEPAGE 'utf-8' CONTENTTYPE 'application/json';

// 
	INPUT = SYS_HTTPPOSTRESPONSE;
	OUTPUT = CONVERTXMLJSON(INPUT,'XML');
	READXMLSTRUCTURE OUTPUT TO TRANSX;
// 
	CLEARTABLE TRANSX WHERE TRANSX_NAME != 'detected_source_language' && TRANSX_NAME != 'text';
// 

	LOOP AT TRANSX 
	BEGIN

		IF TRANSX_NAME == 'detected_source_language' 
				&& TRANSX_VALUE != '' THEN
			MOVE TRANSX_VALUE TO HAKTBLTRANS_LANG;
		ENDIF;


		IF TRANSX_NAME == 'text' 
				&& TRANSX_VALUE != '' THEN
			MOVE TRANSX_VALUE TO HAKTBLTRANS_TEXT;
		ENDIF;

	ENDLOOP;

ENDIF;

 RETURN HAKTBLTRANS_TEXT;
